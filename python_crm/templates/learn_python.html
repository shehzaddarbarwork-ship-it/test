Classes and Objects

# Class definition
class Company:
    def __init__(self, name, sector):
        self.name = name
        self.sector = sector
    
    def get_info(self):
        return f"{self.name} operates in {self.sector}"

# Creating an object
company = Company("TechCorp", "Technology")
print(company.get_info())

Web Development with Flask

Simple Flask Application

from flask import Flask, jsonify

app = Flask(__name__)

@app.route('/')
def home():
    return "Welcome to Python CRM!"

@app.route('/api/users')
def get_users():
    users = [
        {"id": 1, "name": "Alice"},
        {"id": 2, "name": "Bob"}
    ]
    return jsonify(users)

if __name__ == '__main__':
    app.run(debug=True)

Database Operations with SQLAlchemy

Model Definition

from flask_sqlalchemy import SQLAlchemy

db = SQLAlchemy()

class User(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(50), nullable=False)
    email = db.Column(db.String(100), nullable=False)
    
    def to_dict(self):
        return {
            'id': self.id,
            'username': self.username,
            'email': self.email
        }
